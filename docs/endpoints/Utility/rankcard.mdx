# Rank Card
A cool rank card for your leveling system!
## How to use
#### Base URL 
```js
https://api.aggelos-007.xyz/rankcard
```
### Fields
import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs>
  <TabItem value="required" label="Required">
    - avatar
        - Type: Image URL
    - username 
      - Type: String
    - xp
        - Type: Number
    - maxxp
        - Type: Number
    - level
        - Type: Number
  </TabItem>
  <TabItem value="optional" label="Optional">
    - background 
        - Type: Image URL
    - blur 
        - Type: Boolean
    - xpcolor 
        - Type: Hex
    - fontcolor 
        - Type: Hex
    - bgcolor 
        - Type: Hex
    - border
        - Type: Boolean
:::note
    The `xpcolor`, `fontcolor` & `bgcolor` fields should **not** contain **#**
:::
    </TabItem>
</Tabs>
