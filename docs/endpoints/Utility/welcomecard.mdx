# Rank Card
A cool rank card for your leveling system!
## How to use
#### Base URL 
```js
https://api.aggelos-007.xyz/rankcard
```
### Fields
import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs>
  <TabItem value="required" label="Required">
    - avatar
        - Type: Image URL
  </TabItem>
  <TabItem value="optional" label="Optional">
    - text1
        - Type: String
    - text2
        - Type: String
    - text3
        - Type: String
    - background 
        - Type: Image URL
    - blur 
        - Type: Boolean
    - bgcolor 
        - Type: Hex
    - fontcolor
        -  Type: Hex
    - border
        - Type: Boolean
    - bordercolor
        - Type: Hex
:::note
    The `fontcolor`, `bordercolor` & `bgcolor` fields should **not** contain **#**
:::
    </TabItem>
</Tabs>
